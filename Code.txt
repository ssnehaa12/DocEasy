# Doctor Appointment System
A web application allowing patients to book appointments with doctors. Built with PHP and MySQL, styled with HTML and CSS.

## Features
- User registration and login
- Book appointments
- View appointments

## Tech Stack
- PHP
- MySQL
- HTML/CSS

## Installation
1. Clone the repository.
2. Import `doctor_appointment_db.sql` into your MySQL database.
3. Configure database credentials in `db.php`.
4. Run the application on a local server using XAMPP or similar.

CREATE DATABASE doctor_appointment_db;

USE doctor_appointment_db;

-- Table for storing user data
CREATE TABLE users (
    userid INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    age INT NOT NULL,
    email VARCHAR(50) UNIQUE NOT NULL,
    phone VARCHAR(15) NOT NULL,
    password VARCHAR(255) NOT NULL
);

-- Table for login credentials
CREATE TABLE login (
    userid INT PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    password VARCHAR(255) NOT NULL,
    FOREIGN KEY (userid) REFERENCES users(userid)
);

-- Table for storing appointment details
CREATE TABLE appointments (
    appointment_id INT AUTO_INCREMENT PRIMARY KEY,
    userid INT,
    doctor_name VARCHAR(50),
    specialty VARCHAR(50),
    slot VARCHAR(50),
    FOREIGN KEY (userid) REFERENCES users(userid)
);

-- Table for contact messages
CREATE TABLE contact (
    contact_id INT AUTO_INCREMENT PRIMARY KEY,
    address VARCHAR(255),
    email VARCHAR(50),
    phone VARCHAR(15),
    message TEXT
);


<?php
$servername = "localhost";

$password = "";
$dbname = "doctor_appointment_db";

// Create connection

<?php
include 'db.php';

if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    $username = $_POST['username'];
    $age = $_POST['age'];
    $email = $_POST['email'];
    $phone = $_POST['phone'];
    $password = password_hash($_POST['password'], PASSWORD_BCRYPT);

    $sql = "INSERT INTO users (username, age, email, phone, password) 
            VALUES ('$username', $age, '$email', '$phone', '$password')";

    if ($conn->query($sql) === TRUE) {
        echo "User registered successfully!";
    } else {
        echo "Error: " . $conn->error;
    }

    $conn->close();
}
?>


<?php
session_start();
include 'db.php';

if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    $username = $_POST['username'];
    $password = $_POST['password'];

    $sql = "SELECT * FROM users WHERE username='$username'";
    $result = $conn->query($sql);

    if ($result->num_rows > 0) {
        $row = $result->fetch_assoc();
        if (password_verify($password, $row['password'])) {
            $_SESSION['userid'] = $row['userid'];
            header("Location: book_appointment.php");
        } else {
            echo "Invalid password.";
        }
    } else {
        echo "No user found.";
    }

    $conn->close();
}
?>
